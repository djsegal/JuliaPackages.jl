<p dir="auto">StippleLatex uses Vue-Katex to imlement LaTeX-Formatting in Stipple.
There are three possibilities of entering LaTeX content in your web page</p>
<ul dir="auto">
<li>
<p dir="auto">Latex-span element.</p>
<p dir="auto"><code>latex(&lt;LaTeX formula&gt;, &lt;formatting options&gt;)</code></p>
</li>
<li>
<p dir="auto">HTML element with latex content via string macro with an optional modifier which can take the values <code>auto</code>or <code>display</code></p>
<p dir="auto"><code>span(latex"&lt;LaTeX formula&gt;")</code><br>
<code>span(latex"&lt;LaTeX formula&gt;"display)</code></p>
</li>
<li>
<p dir="auto">HTML element with latex content via <code>@latex</code> macro with additional options</p>
<p dir="auto"><code>span(@latex(raw"&lt;LaTeX formula&gt;")</code><br>
<code>span(@latex(raw"&lt;LaTeX formula&gt;", display = true)</code></p>
</li>
</ul>
<p dir="auto">All arguments also support Symbols to bind to model fields.
Here's a demo app that shows possible use cases.</p>

using StippleUI
using StippleLatex

# define a small formula generator
function nestlist(f, a; init = nothing)
    T = eltype(a)
    list = T[]
    el = init
    for (i, x) in enumerate(a)
        el = i == 1 &amp;&amp; init === nothing ? x : f(el, x)
        push!(list, el)
    end
    list
end

formula = nestlist(*, [&quot;&quot;, raw&quot;\sin&quot;, &quot;^2&quot;, &quot; x&quot;, &quot; +&quot;, raw&quot; \sqrt{&quot;, &quot;a&quot;, &quot;^2&quot;, &quot; +&quot;, &quot; b&quot;, &quot;^2&quot;])
formula[contains.(formula, &quot;sqrt&quot;)] .*= &quot;}&quot;

# setting up the app
@app begin
    @in x = 0
    @in formula_1 = raw&quot;\int_{a}^{b} f(x) \, dx = F(x)\Biggr|^b_a&quot;
    @in formula_2 = raw&quot;&quot;
    @private p = @task 1 + 1

    @onchange isready begin
        if !istaskstarted(p) || istaskdone(p) 
            p = @task begin
                println(&quot;Task started&quot;)
                while x &lt;= 100
                    sleep(1)
                    x += 1
                    pos = x &lt; 6 ? 1 : (x - 5) % (length(formula) + 5) + 1
                    formula_2 = formula[min(pos, length(formula))]
                end
            end
            schedule(p)
        end
    end
end

function ui()
    [
        row(cell(class = &quot;st-module&quot;, [
            cell(h1(latex(&quot;\\LaTeX&quot;) * &quot;-Demo&quot;))
            cell(h2(latex&quot;a^2 + b^2 = c^2&quot;))
        ]))

        row(cell(class = &quot;st-module&quot;, [
            textfield(&quot;Enter your LaTeX-Forumla&quot;, :formula_1,)
            cell(class = &quot;q-pa-md&quot;, latex&quot;:formula_1&quot;display)
            row([
                cell(class = &quot;q-pa-md bg-red-1&quot;, raw&quot;&quot;&quot;cell(latex&quot;\cos^2x&quot;display)&quot;&quot;&quot;)
                cell(class = &quot;q-pa-md bg-green-1&quot;, latex&quot;\cos^2x&quot;display)
            ])
            row([
                cell(class = &quot;q-pa-md bg-red-1&quot;, raw&quot;&quot;&quot;cell(latex&quot;This is auto mode with a formula \(\cos^2x\)&quot;auto)&quot;&quot;&quot;)
                cell(class = &quot;q-pa-md bg-green-1&quot;, latex&quot;This is auto mode with a formula \(\cos^2x\)&quot;auto)
            ])
            row([
                cell(class = &quot;q-pa-md bg-red-1&quot;, raw&quot;&quot;&quot;latex(class = &quot;q-pa-md&quot;, raw&quot;\tan^2x&quot;, display = true)&quot;&quot;&quot;)
                cell(class = &quot;bg-green-1 q-pa-md&quot;, latex(class = &quot;q-pa-md&quot;, raw&quot;\tan^2x&quot;, display = true))
            ])
            bignumber(&quot;Wait for 5&quot;, :x, color = R&quot;x &gt;= 5 ? 'negative' : 'positive'&quot;, icon = &quot;calculate&quot;)
            
            row([
                cell(class = &quot;q-pa-md bg-red-1&quot;, raw&quot;&quot;&quot;cell(class = &quot;q-pa-md&quot;, @latex(raw&quot;\tanh^2 y&quot;, display = R&quot;x &gt;= 5&quot;))&quot;&quot;&quot;)
                cell(class = &quot;q-pa-md bg-green-1&quot;, @latex(raw&quot;\tanh^2 y&quot;, display = R&quot;x &gt;= 5&quot;))
            ])


            row([
                cell(class = &quot;q-pa-md bg-red-1&quot;, raw&quot;&quot;&quot;@latex(raw&quot;This is auto mode with an inline formula \(\cos^2x\) and a display formula $$\sin^2x$$&quot;&quot;&quot;)
                cell(class = &quot;q-pa-md bg-green-1&quot;, @latex(raw&quot;This is auto mode with an inline formula \(\cos^2x\) and a display formula $$\sin^2x$$&quot;, auto = true))
            ])
        ]))

        row(cell(class = &quot;st-module&quot;, [
            textfield(class = &quot;q-pa-lg&quot;, &quot;LaTeX&quot;, :formula_2)
            cell(class = &quot;q-pa-md&quot;, &quot;Result:&quot;)
            cell(class = &quot;q-pa-md&quot;, latex&quot;:formula_2&quot;display)
        ]))
    ]
end

route(&quot;/&quot;) do
    page(@init(), ui()) |&gt; html
