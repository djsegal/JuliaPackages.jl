<p dir="auto">This package is a Julia wrapper for the
<a href="https://astro-informatics.github.io/ssht/" rel="nofollow">SSHT</a> library. It
calculates spin-weighted spherical harmonic transforms.</p>

L = lmax + 1
nphi = SSHT.sampling_dh_nphi(L)

thetas = [SSHT.sampling_dh_t2theta(t, L) for t in 1:ntheta];

flm = zeros(Complex{Float64}, L^2);
flm[SSHT.sampling_elm2ind(1, +1)] = 1;

f = SSHT.core_dh_inverse_sov(flm, L, +1);
