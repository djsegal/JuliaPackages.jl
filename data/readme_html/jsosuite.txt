x0 = [-1.2; 1.0]
f = x -&gt; 100 * (x[2] - x[1]^2)^2 + (x[1] - 1)^2
stats = solve(f, x0)

# Constrained problem
c = x -&gt; [x[1] * x[2] - 1]
stats = solve(f, x0, c, [0.0], [0.0])

# Constrained problem in Float32
stats = solve(f, Float32.(x0), c, Float32[0.0], Float32[0.0])"><pre><span class="pl-k">using</span> JSOSuite
