<span class="pl-s">           0.3,</span>
<span class="pl-s">           0.3,</span>
<span class="pl-s">           0.3</span>
<span class="pl-s">         ],</span>
<span class="pl-s">         "min": [</span>
<span class="pl-s">           -0.3,</span>
<span class="pl-s">           -0.3,</span>
<span class="pl-s">           -0.3</span>
<span class="pl-s">         ],</span>
