<p dir="auto">You're in a Jupyter notebook, and have some <code>things</code>.</p>
<p dir="auto">You would like to make a plot based on each <code>thing</code>:</p>
<div class="highlight highlight-source-julia notranslate position-relative overflow-auto" dir="auto" data-snippet-clipboard-copy-content="using PlotIter

# Example data

things = [
    (; title=&quot;A&quot;, w=1), (; title=&quot;B&quot;, w=2), (; title=&quot;C&quot;, w=3), 
    (; title=&quot;D&quot;, w=4), (; title=&quot;E&quot;, w=5), 
];

# Make some plots!
plot_iter(things; ylims_convex_hull=true) do thing
    plot!(x, sin.(x) .* thing.w; title=thing.title)
end;"><pre><span class="pl-k">using</span> PlotIter
