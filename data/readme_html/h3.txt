<p dir="auto"><code>H3.jl</code> provides a Julia version of H3, Hexagonal hierarchical geospatial indexing system. <a href="https://github.com/uber/h3">https://github.com/uber/h3</a></p>
<div class="highlight highlight-source-julia notranslate position-relative overflow-auto" dir="auto" data-snippet-clipboard-copy-content="using H3.API

base = geoToH3(GeoCoord(deg2rad(0), deg2rad(0)), 5)
rings = kRing(base, 1)

x = Vector{Float64}()
y = Vector{Float64}()
for boundary in h3ToGeoBoundary.(rings), geo in boundary
    push!(x, geo.lon)
    push!(y, geo.lat)
end

using UnicodePlots
