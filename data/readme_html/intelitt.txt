<p dir="auto">You need to set the environment variable <code>ENABLE_JITPROFILING=1</code>. Also note that for Julia &lt;= 1.8, you need to compile Julia locally with <code>USE_INTEL_JITEVENTS=1</code>.</p>
<p dir="auto">I recommend using "Start Paused" to minimize the amount of noise
you see due to startup and compilation.</p>

@assert IntelITT.available()

function profile_test(n)
    for i = 1:n
        A = randn(100,100,20)
        m = maximum(A)
        Am = mapslices(sum, A; dims=2)
        B = A[:,:,5]
        Bsort = mapslices(sort, B; dims=1)
        b = rand(100)
        C = B.*b
    end
end

# Compile code once
profile_test(1)

# Resume profiling
IntelITT.resume()
profile_test(100)
