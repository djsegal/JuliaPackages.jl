<pre><code>using LibBSON

bsonObject = BSONObject(Dict(
    "null" =&gt; nothing,
    "bool" =&gt; true,
    "int" =&gt; 42,
    "double" =&gt; 3.141,
    "string" =&gt; "Hello, Jérôme",
    "oid" =&gt; BSONOID(),
    "minkey" =&gt; :minkey,
    "maxkey" =&gt; :maxkey,
      "binaryData" =&gt; Array{UInt8}(20),
    "array" =&gt; Any[5.41, false]
    ))
println(bsonObject)
println(bsonObject["string"])
for (k, v) in bsonObject
    println("$k =&gt; $v")
end
bsonArray = BSONArray(Any[
    "one",
    Dict("key" =&gt; 6.7)
    ])
for e in bsonArray
    println(e)
end
</code></pre>
